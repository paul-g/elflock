20  	private List<TableColumn<T, String>> makeColumns(final Class<T> clazz) {
21  		final List<TableColumn<T, String>> columns = new ArrayList<TableColumn<T, String>>();
22  		for (final Field f : clazz.getDeclaredFields()) {
23  			final TableColumn<T, String> column = new TableColumn<T, String>(f.getName());
24  			column.setCellValueFactory(new PropertyValueFactory<T, String>(f.getName()));
25  			columns.add(column);
