106 	private ObservableList<Data> toNegativePieChartData() {
107 		double total = (totalSpent == null ? 7500 : totalSpent);
108 		double leftTotal = total;
109 		pieChartNegData.clear();
110 		for (AggregatedRecord record : groupData) {
111 			pieChartNegData.add(new PieChart.Data(record.getDescription(),
112 					(Math.abs(record.getNegative()) / total) * 100));
113 			leftTotal -= record.getNegative();
114 		}
115 		pieChartNegData.add(new PieChart.Data("Other", (leftTotal / total) * 100));
116 		return pieChartNegData;
117 	}

119 	private ObservableList<Data> toPositivePieChartData() {
120 		double total = (totalIncome == null ? 7500 : totalIncome);
121 		double leftTotal = total;
122 		pieChartPosData.clear();
123 		for (AggregatedRecord record : groupData) {
124 			pieChartPosData.add(new PieChart.Data(record.getDescription(), (record.getPositive() / total) * 100));
125 			leftTotal -= record.getPositive();
126 		}
127 		pieChartPosData.add(new PieChart.Data("Other", (leftTotal / total) * 100));
128 		return pieChartPosData;

226 	public void fileSelected(final String path) throws IOException {
227 		final HistoryFileVisitor fileVisitor = new HistoryFileVisitor();
228 		recordStore = fileVisitor.getRecordStore();
229 
230 		Files.walkFileTree(Paths.get(path), fileVisitor);
231 		groupBy.setDisable(false);
232 		search.setDisable(false);
233 		totalSpent = (int) recordStore.getTotalSpent();
234 		totalIncome = (int) recordStore.getTotalIncome();
235 		data.clear();
236 		data.addAll(recordStore.getAllRecords());
237 		accountsData.clear();
238 		accountsData.addAll(recordStore.getAccounts());
239 		recordStore.printSummary();
240 	}
